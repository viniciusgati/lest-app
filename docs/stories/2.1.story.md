# Story 2.1 — Backend — Matérias & Temas (Models + API)

## Status: Ready for Review

## Story

**Como** usuário do EduTrack,
**Quero** que o backend gerencie minhas matérias e temas,
**Para que** eu possa organizar meu conteúdo de estudo de forma hierárquica e pessoal.

---

## Contexto

Primeira story do Epic 2. O backend Rails está pronto com auth funcionando (Epic 1). Esta story cria os models `Subject` (Matéria) e `Topic` (Tema), as migrations, e os endpoints REST protegidos por autenticação.

Cada usuário possui suas próprias matérias e temas — nenhum dado é compartilhado entre usuários. O `Topic` já carrega os campos do algoritmo SM-2 (`ease_factor`, `interval`, `next_review`) e o campo de nota rápida (`notes`), que serão usados em stories futuras.

**Dependência:** Epic 1 completo (Stories 1.1 a 1.5).

### SubjectsController

```ruby
# app/controllers/api/v1/subjects_controller.rb
module Api
  module V1
    class SubjectsController < BaseController
      before_action :set_subject, only: [:update, :destroy]

      def index
        subjects = current_user.subjects.order(:name)
        render json: subjects
      end

      def create
        subject = current_user.subjects.build(subject_params)
        if subject.save
          render json: subject, status: :created
        else
          render json: { errors: subject.errors.full_messages }, status: :unprocessable_entity
        end
      end

      def update
        if @subject.update(subject_params)
          render json: @subject
        else
          render json: { errors: @subject.errors.full_messages }, status: :unprocessable_entity
        end
      end

      def destroy
        @subject.destroy
        head :no_content
      end

      private

      def set_subject
        @subject = current_user.subjects.find(params[:id])
      rescue ActiveRecord::RecordNotFound
        render json: { error: 'Not found' }, status: :not_found
      end

      def subject_params
        params.require(:subject).permit(:name)
      end
    end
  end
end
```

### TopicsController

```ruby
# app/controllers/api/v1/topics_controller.rb
module Api
  module V1
    class TopicsController < BaseController
      before_action :set_subject
      before_action :set_topic, only: [:update, :destroy]

      def index
        render json: @subject.topics.order(:name)
      end

      def create
        topic = @subject.topics.build(topic_params)
        if topic.save
          render json: topic, status: :created
        else
          render json: { errors: topic.errors.full_messages }, status: :unprocessable_entity
        end
      end

      def update
        if @topic.update(topic_params)
          render json: @topic
        else
          render json: { errors: @topic.errors.full_messages }, status: :unprocessable_entity
        end
      end

      def destroy
        @topic.destroy
        head :no_content
      end

      private

      def set_subject
        @subject = current_user.subjects.find(params[:subject_id])
      rescue ActiveRecord::RecordNotFound
        render json: { error: 'Not found' }, status: :not_found
      end

      def set_topic
        @topic = @subject.topics.find(params[:id])
      rescue ActiveRecord::RecordNotFound
        render json: { error: 'Not found' }, status: :not_found
      end

      def topic_params
        params.require(:topic).permit(:name, :notes)
      end
    end
  end
end
```

### Specs dos endpoints

```ruby
# spec/requests/api/v1/subjects_spec.rb
require 'rails_helper'

RSpec.describe 'Api::V1::Subjects', type: :request do
  let(:user) { create(:user) }
  let(:token) { user_token(user) } # helper que retorna JWT
  let(:headers) { { 'Authorization' => "Bearer #{token}" } }

  describe 'GET /api/v1/subjects' do
    it 'retorna matérias do usuário' do
      create_list(:subject, 3, user: user)
      get '/api/v1/subjects', headers: headers
      expect(response).to have_http_status(:ok)
      expect(JSON.parse(response.body).length).to eq(3)
    end

    it 'não retorna matérias de outro usuário' do
      create(:subject) # outro usuário
      get '/api/v1/subjects', headers: headers
      expect(JSON.parse(response.body).length).to eq(0)
    end

    it 'retorna 401 sem token' do
      get '/api/v1/subjects'
      expect(response).to have_http_status(:unauthorized)
    end
  end

  describe 'POST /api/v1/subjects' do
    it 'cria matéria' do
      post '/api/v1/subjects',
           params: { subject: { name: 'Matemática' } },
           headers: headers, as: :json
      expect(response).to have_http_status(:created)
    end

    it 'retorna erro sem name' do
      post '/api/v1/subjects',
           params: { subject: { name: '' } },
           headers: headers, as: :json
      expect(response).to have_http_status(:unprocessable_entity)
    end
  end

  describe 'DELETE /api/v1/subjects/:id' do
    it 'remove matéria do usuário' do
      subject = create(:subject, user: user)
      delete "/api/v1/subjects/#{subject.id}", headers: headers
      expect(response).to have_http_status(:no_content)
    end

    it 'retorna 404 para matéria de outro usuário' do
      other_subject = create(:subject)
      delete "/api/v1/subjects/#{other_subject.id}", headers: headers
      expect(response).to have_http_status(:not_found)
    end
  end
end
```

### Helper de autenticação para specs

```ruby
# spec/support/auth_helpers.rb
module AuthHelpers
  def user_token(user)
    post '/api/v1/auth/login',
         params: { user: { email: user.email, password: user.password } },
         as: :json
    response.headers['Authorization']&.split(' ')&.last
  end
end

RSpec.configure do |config|
  config.include AuthHelpers, type: :request
end
```
